# 基本介紹
- PHP 小工具

# 環境
- php 5

# 功能介紹
- js_locked_right_click() = 網頁鎖右鍵
- js_alert() = 網頁訊息談視窗
- data_encode() = 加密
- data_decode() = 解密
- get_token() = 取得 Token
- check_token() = Token驗證(得配合session使用)
- get_user_ip() = 取得使用者ip
- random_string() = 亂數產生
- pre_p = html <pre> + php print_r() + html <hr>
- pre_v = html <pre> + php var_dump() + html <hr>
- check_date = 日期檢查
- check_phone = 台灣手機號碼檢查
- check_identity = 台灣身分證檢查
